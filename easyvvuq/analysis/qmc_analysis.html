111    def contains_nan(self, values):
112        """
113        Checks if ``None`` or ``numpy.nan`` exists in `values`. Returns ``True`` if
114        any there are at least one occurrence of ``None`` or ``numpy.nan``.
115        Parameters
116        ----------
117        values : array_like, list, number
118            `values` where to check for occurrences of ``None`` or ``np.nan``.
119            Can be irregular and have any number of nested elements.
120        Returns
121        -------
122        bool
123            ``True`` if `values` has at least one occurrence of ``None`` or
124            ``numpy.nan``.
125        """
126        # To speed up we first try the fast option np.any(np.isnan(values))
127        try:
128            return np.any(np.isnan(values))
129        except (ValueError, TypeError):
130            if values is None or values is np.nan:
131                return True
132            # To solve the problem of float/int as well as numpy int/flaot
133            elif np.isscalar(values) and np.isnan(values):
134                return True
135            elif hasattr(values, "__iter__"):
136                for value in values:
137                    if self.contains_nan(value):
138                        return True
139
140                return False
141            else:
142                return False

144    def create_mask(self, samples):
145        """
146        Mask samples that do not give results (anything but np.nan or None).
147        Parameters
148        ----------
149        samples : array_like
150            Evaluations for the model.
151        Returns
152        -------
153        masked_samples : list
154            The evaluations that have results (not numpy.nan or None).
155        mask : boolean array
156            The mask itself, used to create the masked arrays.
157        """
158        masked_samples = []
159        mask = np.ones(len(samples), dtype=bool)
160
161        for i, result in enumerate(samples):
162            # if np.any(np.isnan(result)):
163            if self.contains_nan(result):
164                mask[i] = False
165            else:
166                masked_samples.append(result)
167
168        return masked_samples, mask

170    def analyse(self, data_frame):
171        """Perform QMC analysis on a given pandas DataFrame.
172
173        Parameters
174        ----------
175        data_frame : pandas DataFrame
176            Input data for analysis.
177
178        Returns
179        -------
180        easyvvuq.analysis.qmc.QMCAnalysisResults
181            AnalysisResults object for QMC.
182        """
183        if data_frame.empty:
184            raise RuntimeError(
185                "No data in data frame passed to analyse element")
186
187        qoi_cols = self.qoi_cols
188
189        results = {
190            'statistical_moments': {k: {} for k in qoi_cols},
191            'percentiles': {k: {} for k in qoi_cols},
192            'sobols_first': {k: {} for k in qoi_cols},
193            'sobols_total': {k: {} for k in qoi_cols},
194            'conf_sobols_first': {k: {} for k in qoi_cols},
195            'conf_sobols_total': {k: {} for k in qoi_cols}
196        }
197
198        # Extract output values for each quantity of interest from Dataframe
199        samples = self.get_samples(data_frame)
200
201        # Compute descriptive statistics for each quantity of interest
202        for k in qoi_cols:
203            # Find NaNs and create a mask excluding these samples from the analysis
204            # https://github.com/simetenn/uncertainpy/blob/ffb2400289743066265b9a8561cdf3b72e478a28/src/uncertainpy/core/uncertainty_calculations.py#L1532
205            masked_samples, mask = self.create_mask(samples[k])
206
207            results['statistical_moments'][k] = {'mean': np.mean(masked_samples, axis=0),
208                                                 'var': np.var(masked_samples, axis=0),
209                                                 'std': np.std(masked_samples, axis=0)}
210            results['percentiles'][k] = {'p10': np.percentile(masked_samples, 10, 0)[0],
211                                         'p50': np.percentile(masked_samples, 50, 0)[0],
212                                         'p90': np.percentile(masked_samples, 90, 0)[0]}                                                 
213
214            # Replace Nan values by the mean before proceeding with the SA
215            indices = np.where(mask == 0)[0] # samples[~mask] = results[k].mean
216            for i in indices:
217                samples[k][i] = results['statistical_moments'][k]['mean']
218
219            if not np.all(mask):
220                print("Warning: QoI \"{}\" only yields ".format(k) +
221                    "results for {}/{} ".format(sum(mask), len(mask)) +
222                    "parameter combinations. " +
223                    "Runs {} are not valid. ".format(indices+1) +
224                    "NaN results are set to the mean when calculating the Sobol indices. " +
225                    "This might affect the Sobol indices.")
226            
227            sobols_first, conf_first, sobols_total, conf_total = \
228                self.sobol_bootstrap(samples[k])
229            results['sobols_first'][k] = sobols_first
230            results['sobols_total'][k] = sobols_total
231            results['conf_sobols_first'][k] = conf_first
232            results['conf_sobols_total'][k] = conf_total
233
234        return QMCAnalysisResults(raw_data=results, samples=data_frame,
235                                  qois=self.qoi_cols, inputs=list(self.sampler.vary.get_keys()))

238        """

259    def sobol_bootstrap(self, samples, alpha=0.05, n_samples=1000):
260        """
261        Computes the first order and total order Sobol indices using Saltelli's
262        method. To assess the sampling inaccuracy, bootstrap confidence intervals
263        are also computed.
264
265        Reference: A. Saltelli, Making best use of model evaluations to compute
266        sensitivity indices, Computer Physics Communications, 2002.
267
268        Parameters
269        ----------
270        samples : list
271            The samples for a given QoI.
272        alpha: float
273            The (1 - alpha) * 100 confidence interval parameter. The default is 0.05.
274        n_samples: int
275            The number of bootstrap samples. The default is 1000.
276
277        Returns
278        -------
279        sobols_first_dict, conf_first_dict, sobols_total_dict, conf_total_dict:
280        dictionaries containing the first- and total-order Sobol indices for all
281        parameters, and (1-alpha)*100 lower and upper confidence bounds.
282
283        """
284        assert len(samples) > 0
285        assert alpha > 0.0
286        assert alpha < 1.0
287        assert n_samples > 0
288
289        # convert to array
290        samples = np.array(samples)
291        # the number of parameter and the number of MC samples in n_mc * (n_params + 2)
292        # and the size of the QoI
293        n_params = self.sampler.n_params
294        n_mc = self.sampler.n_mc_samples
295        n_qoi = samples[0].size
296        sobols_first_dict = {}
297        conf_first_dict = {}
298        sobols_total_dict = {}
299        conf_total_dict = {}
300
301        for j, param_name in enumerate(self.sampler.vary.get_keys()):
302            # code evaluations of input matrices M1, M2 and Ni, i = 1,...,n_params
303            # see reference above.
304            f_M2, f_M1, f_Ni = self._separate_output_values(samples, n_params, n_mc)
305            # our point estimate for the 1st and total order Sobol indices
306            value_first = self._first_order(f_M2, f_M1, f_Ni[:, j])
307            value_total = self._total_order(f_M2, f_M1, f_Ni[:, j])
308            # array for resampled estimates
309            sobols_first = np.zeros([n_samples, n_qoi])
310            sobols_total = np.zeros([n_samples, n_qoi])
311            for i in range(n_samples):
312                # resample, must be done on already seperated output due to
313                # the specific order in samples
314                idx = np.random.randint(0, n_mc - 1, n_mc)
315                f_M2_resample = f_M2[idx]
316                f_M1_resample = f_M1[idx]
317                f_Ni_resample = f_Ni[idx]
318                # recompute Sobol indices
319                sobols_first[i] = self._first_order(f_M2_resample, f_M1_resample,
320                                                    f_Ni_resample[:, j])
321                sobols_total[i] = self._total_order(f_M2_resample, f_M1_resample,
322                                                    f_Ni_resample[:, j])
323            # compute confidence intervals
324            _, low_first, high_first = confidence_interval(sobols_first, value_first,
325                                                           alpha, pivotal=True)
326            _, low_total, high_total = confidence_interval(sobols_total, value_total,
327                                                           alpha, pivotal=True)
328            # store results
329            sobols_first_dict[param_name] = value_first
330            conf_first_dict[param_name] = {'low': low_first, 'high': high_first}
331            sobols_total_dict[param_name] = value_total
332            conf_total_dict[param_name] = {'low': low_total, 'high': high_total}
333
334        return sobols_first_dict, conf_first_dict, sobols_total_dict, conf_total_dict

337    @staticmethod
338    def _separate_output_values(samples, n_params, n_mc_samples):
339        """There are n_params + 2 different input matrices: M1, M2, N_i,
340        i=1,...,n_params.  (see reference under sobol_bootstrap). The
341        EasyVVUQ dataframe is stored in the order:
342
343        [sample from M2, sample from N1, N2, ... sample from N_n_params,
344         sample from M1, repeat].
345
346        This subroutine separates the output values into the contributions
347        of the different input matrices.
348
349        Parameters
350        ----------
351        samples: list
352            The samples for a given QoI
353        n_params: int
354            The number of uncertain input parameters.
355        n_mc_samples: int
356            The number of MC samples per input matrix, i.e. the
357          number of rows in M1, M2 or Ni.
358
359        Returns
360        -------
361        NumPy arrays of the separated code evaluations: f_M2, f_M1, f_Ni, where
362        f_Ni contains n_params entries corresponding to the n_params Ni matrices.
363
364        """
365        evaluations = np.array(samples)
366
367        shape = (n_mc_samples, n_params) + evaluations[0].shape
368        step = n_params + 2
369        f_Ni = np.zeros(shape)
370
371        f_M2 = evaluations[0:evaluations.shape[0]:step]
372        f_M1 = evaluations[(step - 1):evaluations.shape[0]:step]
373
374        for i in range(n_params):
375            f_Ni[:, i] = evaluations[(i + 1):evaluations.shape[0]:step]
376
377        return f_M2, f_M1, f_Ni

379    @staticmethod
380    def _first_order(f_M2, f_M1, f_Ni):
381        """Calculate first order sensitivity indices.
382
383        Parameters
384        ----------
385        f_M2: NumPy array
386            Array of code evaluations on input array M2
387        f_M1: NumPy array
388            Array of code evaluations on input array M1
389        f_Ni: NumPy array
390            Array of code evaluations on input array Ni, i=1,...,n_params
391
392        Returns
393        -------
394        A NumPy array of the n_params first-order Sobol indices.
395        """
396        V = np.var(np.r_[f_M2, f_M1], axis=0)
397        return np.mean(f_M1 * (f_Ni - f_M2), axis=0) / (V + (V == 0)) * (V != 0)
398
399    @staticmethod
400    def _total_order(f_M2, f_M1, f_Ni):
401        """Calculate total order sensitivity indices. See also:
402
403        A Saltelli et al, Variance based sensitivity analysis of model output.
404        Design and estimator for the total sensitivity index, 2009.
405
406        Parameters
407        ----------
408        f_M2: NumPy array
409            Array of code evaluations on input array M2 (matrix A in ref above)
410        f_M1: NumPy array
411            Array of code evaluations on input array M1 (matrix B in ref above)
412        f_Ni: NumPy array
413            Array of code evaluations on input array Ni, i=1,...,n_params
414          (matrix AB in ref above)
415
416        Returns
417        -------
418        A NumPy array of the n_params total-order Sobol indices.
419        """
420        V = np.var(np.r_[f_M2, f_M1], axis=0)
421        return 0.5 * np.mean((f_M2 - f_Ni) ** 2, axis=0) / (V + (V == 0)) * (V != 0)

260    def sobol_bootstrap(self, samples, alpha=0.05, n_samples=1000):
261        """
262        Computes the first order and total order Sobol indices using Saltelli's
263        method. To assess the sampling inaccuracy, bootstrap confidence intervals
264        are also computed.
265
266        Reference: A. Saltelli, Making best use of model evaluations to compute
267        sensitivity indices, Computer Physics Communications, 2002.
268
269        Parameters
270        ----------
271        samples : list
272            The samples for a given QoI.
273        alpha: float
274            The (1 - alpha) * 100 confidence interval parameter. The default is 0.05.
275        n_samples: int
276            The number of bootstrap samples. The default is 1000.
277
278        Returns
279        -------
280        sobols_first_dict, conf_first_dict, sobols_total_dict, conf_total_dict:
281        dictionaries containing the first- and total-order Sobol indices for all
282        parameters, and (1-alpha)*100 lower and upper confidence bounds.
283
284        """
285        assert len(samples) > 0
286        assert alpha > 0.0
287        assert alpha < 1.0
288        assert n_samples > 0
289
290        # convert to array
291        samples = np.array(samples)
292        # the number of parameter and the number of MC samples in n_mc * (n_params + 2)
293        # and the size of the QoI
294        n_params = self.sampler.n_params
295        n_mc = self.sampler.n_mc_samples
296        n_qoi = samples[0].size
297        sobols_first_dict = {}
298        conf_first_dict = {}
299        sobols_total_dict = {}
300        conf_total_dict = {}
301
302        for j, param_name in enumerate(self.sampler.vary.get_keys()):
303            # code evaluations of input matrices M1, M2 and Ni, i = 1,...,n_params
304            # see reference above.
305            f_M2, f_M1, f_Ni = self._separate_output_values(samples, n_params, n_mc)
306            # our point estimate for the 1st and total order Sobol indices
307            value_first = self._first_order(f_M2, f_M1, f_Ni[:, j])
308            value_total = self._total_order(f_M2, f_M1, f_Ni[:, j])
309            # array for resampled estimates
310            sobols_first = np.zeros([n_samples, n_qoi])
311            sobols_total = np.zeros([n_samples, n_qoi])
312            for i in range(n_samples):
313                # resample, must be done on already seperated output due to
314                # the specific order in samples
315                idx = np.random.randint(0, n_mc - 1, n_mc)
316                f_M2_resample = f_M2[idx]
317                f_M1_resample = f_M1[idx]
318                f_Ni_resample = f_Ni[idx]
319                # recompute Sobol indices
320                sobols_first[i] = self._first_order(f_M2_resample, f_M1_resample,
321                                                    f_Ni_resample[:, j])
322                sobols_total[i] = self._total_order(f_M2_resample, f_M1_resample,
323                                                    f_Ni_resample[:, j])
324            # compute confidence intervals
325            _, low_first, high_first = confidence_interval(sobols_first, value_first,
326                                                           alpha, pivotal=True)
327            _, low_total, high_total = confidence_interval(sobols_total, value_total,
328                                                           alpha, pivotal=True)
329            # store results
330            sobols_first_dict[param_name] = value_first
331            conf_first_dict[param_name] = {'low': low_first, 'high': high_first}
332            sobols_total_dict[param_name] = value_total
333            conf_total_dict[param_name] = {'low': low_total, 'high': high_total}
334
335        return sobols_first_dict, conf_first_dict, sobols_total_dict, conf_total_dict

338    @staticmethod
339    def _separate_output_values(samples, n_params, n_mc_samples):
340        """There are n_params + 2 different input matrices: M1, M2, N_i,
341        i=1,...,n_params.  (see reference under sobol_bootstrap). The
342        EasyVVUQ dataframe is stored in the order:
343
344        [sample from M2, sample from N1, N2, ... sample from N_n_params,
345         sample from M1, repeat].
346
347        This subroutine separates the output values into the contributions
348        of the different input matrices.
349
350        Parameters
351        ----------
352        samples: list
353            The samples for a given QoI
354        n_params: int
355            The number of uncertain input parameters.
356        n_mc_samples: int
357            The number of MC samples per input matrix, i.e. the
358          number of rows in M1, M2 or Ni.
359
360        Returns
361        -------
362        NumPy arrays of the separated code evaluations: f_M2, f_M1, f_Ni, where
363        f_Ni contains n_params entries corresponding to the n_params Ni matrices.
364
365        """
366        evaluations = np.array(samples)
367
368        shape = (n_mc_samples, n_params) + evaluations[0].shape
369        step = n_params + 2
370        f_Ni = np.zeros(shape)
371
372        f_M2 = evaluations[0:evaluations.shape[0]:step]
373        f_M1 = evaluations[(step - 1):evaluations.shape[0]:step]
374
375        for i in range(n_params):
376            f_Ni[:, i] = evaluations[(i + 1):evaluations.shape[0]:step]
377
378        return f_M2, f_M1, f_Ni

380    @staticmethod
381    def _first_order(f_M2, f_M1, f_Ni):
382        """Calculate first order sensitivity indices.
383
384        Parameters
385        ----------
386        f_M2: NumPy array
387            Array of code evaluations on input array M2
388        f_M1: NumPy array
389            Array of code evaluations on input array M1
390        f_Ni: NumPy array
391            Array of code evaluations on input array Ni, i=1,...,n_params
392
393        Returns
394        -------
395        A NumPy array of the n_params first-order Sobol indices.
396        """
397        V = np.var(np.r_[f_M2, f_M1], axis=0)
398        return np.mean(f_M1 * (f_Ni - f_M2), axis=0) / (V + (V == 0)) * (V != 0)
399
400    @staticmethod
401    def _total_order(f_M2, f_M1, f_Ni):
402        """Calculate total order sensitivity indices. See also:
403
404        A Saltelli et al, Variance based sensitivity analysis of model output.
405        Design and estimator for the total sensitivity index, 2009.
406
407        Parameters
408        ----------
409        f_M2: NumPy array
410            Array of code evaluations on input array M2 (matrix A in ref above)
411        f_M1: NumPy array
412            Array of code evaluations on input array M1 (matrix B in ref above)
413        f_Ni: NumPy array
414            Array of code evaluations on input array Ni, i=1,...,n_params
415          (matrix AB in ref above)
416
417        Returns
418        -------
419        A NumPy array of the n_params total-order Sobol indices.
420        """
421        V = np.var(np.r_[f_M2, f_M1], axis=0)
422        return 0.5 * np.mean((f_M2 - f_Ni) ** 2, axis=0) / (V + (V == 0)) * (V != 0)
