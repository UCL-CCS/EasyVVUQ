105class QCGPJPool(Executor):
106    """A Pool that manages execution of actions with QCG-PilotJob.
107
108    Parameters
109    ----------
110    qcgpj_executor: str
111        An instance of QCGPJExecutor. If not provided, an instance of QCGPJExecutor
112        with default settings will be created
113    template: QCGPJTemplate
114        An object which contains only a single method `template` that returns a tuple.
115        The first element of a tuple should be a string representing a QCG-PilotJob task's description
116        with placeholders (identifiers preceded by $ symbol)
117        and the second a dictionary that assigns default values for selected placeholders.
118        If not provided, a default EasyVVUQBasicTemplate will be used
119    template_params: dict
120        A dictionary that contains parameters that will be used to substitute placeholders
121        defined in the template
122    polling_interval: int
123        An interval between queries to the QCG-PilotJob Manager service about state of the tasks, in seconds.
124    """
125
126    def __init__(
127            self,
128            qcgpj_executor=None,
129            template=None,
130            template_params=None,
131            polling_interval=1):
132        if qcgpj_executor is None:
133            qcgpj_executor = QCGPJExecutor()
134        if template is None:
135            template = EasyVVUQBasicTemplate()
136
137        self._qcgpj_executor = qcgpj_executor
138        self._template = template
139        self._template_params = template_params
140        self._polling_interval = polling_interval
141        self._campaign_dir = None
142
143    def submit(self, fn, *args, **kwargs):
144        """Submits a callable to be executed by QCG-PilotJob.
145
146        Schedules the callable to be executed inside a QCG-PilotJob's task and returns
147        a Future representing the execution of the callable.
148
149        Returns
150        -------
151        QCGPJFuture
152            QCGPJFuture representing the given call.
153        """
154        actions = fn.__self__
155        actions.set_wrapper(QCGPJPool._wrapper)
156        exec = 'python3'
157        arg1 = "-m"
158        arg2 = "easyvvuq.actions.execute_qcgpj_task"
159
160        self._campaign_dir = args[0]['campaign_dir']
161
162        pickled_actions = base64.b64encode(dill.dumps(actions)).decode('ascii')
163        pickled_previous = base64.b64encode(dill.dumps(args[0])).decode('ascii')
164
165        actions_file = f'{self._campaign_dir}/.qcgpj_in_act_{args[0]["run_id"]}'
166        previous_file = f'{self._campaign_dir}/.qcgpj_in_prev_{args[0]["run_id"]}'
167
168        with open(actions_file, 'w') as f:
169            f.write(pickled_actions)
170
171        with open(previous_file, 'w') as f:
172            f.write(pickled_previous)
173
174        return self._qcgpj_executor.submit(
175            self._template.template,
176            self._template_params,
177            exec=exec,
178            name=args[0]['run_id'],
179            stdout=f"{self._campaign_dir}/stdout_{args[0]['run_id']}",
180            stderr=f"{self._campaign_dir}/stderr_{args[0]['run_id']}",
181            args=[arg1, arg2, actions_file, previous_file])
182
183    @property
184    def executor(self):
185        """Returns current QCGPJExecutor instance.
186
187        It gives you an access to QCG-PilotJob Manager instance, which in turn can be used to
188        get information about the QCG-PilotJob execution environment.
189        """
190        return self._qcgpj_executor
191
192    def convert_results(self, result_qcgpj):
193        """Converts results generated by QCG-PilotJob task to EasyVVUQ-compatible form
194
195        The method loads results data from a file where it was stored by QCG-PilotJob's task
196        and then converts it to a dictionary which can be further processed by EasyVVUQ.
197
198        Parameters
199        ----------
200        result_qcgpj: list or None
201            A list of results returned by a QCG-PilotJob task (only the first element will be used),
202            or None if the task hasn't finished with the status SUCCEED
203
204        Returns
205        -------
206        dict
207            A dictionary containing results
208        """
209
210        for key, value in result_qcgpj.items():
211            if value != 'SUCCEED':
212                logging.error(f"Task {key} finished with the status: {value}")
213                raise RuntimeError(f"QCG-PilotJob task {key} finished with the status: {value}")
214
215            result_file = f'{self._campaign_dir}/.qcgpj_result_{key}'
216            with open(result_file, 'r') as f:
217                previous = json.load(f)
218            remove(result_file)
219
220            return previous
221
222    def shutdown(self, **kwargs):
223        """Clean-up the resources associated with the QCGPJPool.
224        """
225        return self._qcgpj_executor.shutdown()

227    def as_completed(self, futures):
228        """Checks for the status of features and yields those that are finished
229        """
230
231        pending = set(futures)
232        finished = set()
233
234        for f in futures:
235            if f.done():
236                finished.add(f)
237        pending = pending - finished
238
239        while finished:
240            yield finished.pop()
241
242        while pending:
243            for f in pending:
244                if f.done():
245                    finished.add(f)
246            pending = pending - finished
247
248            while finished:
249                yield finished.pop()
250
251            time.sleep(self._polling_interval)
